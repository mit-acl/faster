113  vec_Veci<Dim> rayTrace(const Vecf<Dim> &pt1, const Vecf<Dim> &pt2) {
114  Vecf<Dim> diff = pt2 - pt1;
115  decimal_t k = 0.8;
116  int max_diff = (diff / res_).template lpNorm<Eigen::Infinity>() / k;
117  decimal_t s = 1.0 / max_diff;
118  Vecf<Dim> step = diff * s;
119 
120  vec_Veci<Dim> pns;
121  Veci<Dim> prev_pn = Veci<Dim>::Constant(-1);
122  for (int n = 1; n < max_diff; n++) {
123  Vecf<Dim> pt = pt1 + step * n;
124  Veci<Dim> new_pn = floatToInt(pt);
125  if (isOutside(new_pn))
126  break;
127  if (new_pn != prev_pn)
128  pns.push_back(new_pn);
129  prev_pn = new_pn;
130  }
131  return pns;
132  }
